# 성격 유형 검사하기

```
import java.util.*;

class Solution {
    public String solution(String[] survey, int[] choices) {
        String answer = "";
        int temp = 0; //점수 계산을 위한 임시 변수
        //점수 저장할 hashmap 선언 
        HashMap<Character, Integer> map= new HashMap<Character, Integer>(8);
        map.put('R', 0);
        map.put('T', 0);
        map.put('C', 0);
        map.put('F', 0);
        map.put('J', 0);
        map.put('M', 0);
        map.put('A', 0);
        map.put('N', 0);
        
        for(int i=0; i<survey.length; i++) {
            if(choices[i] < 4) {
                temp = 0;
                temp = map.get(survey[i].charAt(0));
                temp = temp + (4-choices[i]);
                //string 함수 중 charAt(index) 번째 문자열 변환
                map.put(survey[i].charAt(0), temp);
            }
            else{
                temp = 0;
                temp = map.get(survey[i].charAt(1));
                temp = temp + (choices[i]-4);
                map.put(survey[i].charAt(1), temp);
            }
        }
        
        if(map.get('R') >= map.get('T')){
            answer += 'R';
        } else{
            answer += 'T';
        }
        if(map.get('C') >= map.get('F')){
            answer += 'C';
        } else{
            answer += 'F';
        }
        if(map.get('J') >= map.get('M')){
            answer += 'J';
        } else{
            answer += 'M';
        }
        if(map.get('A') >= map.get('N')){
            answer += 'A';
        } else{
            answer += 'N';
        }
        
        return answer;
    }
}
```
